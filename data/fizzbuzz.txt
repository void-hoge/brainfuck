fizzarr[5];
fizzarr[0] = 'F';
fizzarr[1] = 'i';
fizzarr[2] = 'z';
fizzarr[3] = 'z';
buzzarr[5];
buzzarr[0] = 'B';
buzzarr[1] = 'u';
buzzarr[2] = 'z';
buzzarr[3] = 'z';
num = 1;
while (num != 0) {
    fizz = num % 3 == 0;
    buzz = num % 5 == 0;
    nofizzbuzz = !(fizz | buzz);
    if (nofizzbuzz) {
        putint(num);
    }
    if (fizz) {
        putarr(fizzarr);
    }
    if (buzz) {
        putarr(buzzarr);
    }
    putchar(' ');
    num += 1;
}
putchar('\n');
