num = 1;
while (num != 0) {
    fizz = num % 3 == 0;
    buzz = num % 5 == 0;
    nofizzbuzz = !(fizz | buzz);
    if (nofizzbuzz) {
        putint(num);
    }
    if (fizz) {
        putchar('F');
        putchar('i');
        putchar('z');
        putchar('z');
    }
    if (buzz) {
        putchar('B');
        putchar('u');
        putchar('z');
        putchar('z');
    }
    putchar('\n');
    num += 1;
}
